; Scheme

(define hello-world
  (lambda ()
    (begin
      (write 'Hello-World)
      (newline)
      (hello-world))))

; Looping version
(define hello-world
  (lambda ()
    (display "Hello, World!")
    (newline)
    (hello-world)))
